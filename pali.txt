DATA_SEG SEGMENT
    STR1 DB 'ABCDCBA'
    LEN DW 7
    PALCHK DB 0
DATA_SEG ENDS
CODE_SEG SEGMENT
ASSUME CS:CODE_SEG,DS:DATA_SEG
    START:MOV AX,DATA_SEG;INITIATE DATA SEGMENT REG
MOV DS,AX
MOV SI,OFFSET STR1
MOV DI,SI
MOV BX,LEN
DEC BX
ADD DI,BX
    AGAIN:MOV AL,[SI]
CMP AL,[DI]
JNE NOPAL
INC SI
DEC DI
CMP SI,DI
JNC AGAIN
MOV dx,0FFH
JMP SKIP
    NOPAL:MOV dx,00H
    SKIP:MOV AH,4CH
INT 21H
CODE_SEG ENDS
END START
