DATA SEGMENT
   NUM DB 01110001B,06H,5BH,4FH,66H,6DH,7DH,07H,7FH,6FH
   COUNT DB 0AH
   TIME DW 0FFFFH
DATA ENDS

CODE SEGMENT PUBLIC 'CODE'
        ASSUME CS:CODE, DS:DATA

START:
   MOV AX, DATA
   MOV DS, AX
   
   MOV AL, 80H
   MOV DX, 06H
   OUT DX, AL
   LEA DI, NUM
   MOV DX, 00H
   
   AGAIN:
      MOV AL, [DI]
      INC DI
      OUT DX, AL
      MOV DX, TIME
   
   DELAY:
      DEC DX
      JNZ DELAY
      DEC COUNT
      JNZ AGAIN
    
ENDLESS:
        JMP ENDLESS
CODE ENDS
        END START
