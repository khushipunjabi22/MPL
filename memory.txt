DATA_SEG SEGMENT
     SRCARY DB 94H, 11H, 22H, 33H, 44H
     DESARY DB 5 DUP(0)
DATA_SEG ENDS
CODE_SEG SEGMENT
ASSUME CS:CODE_SEG,DS:DATA_SEG
     START: MOV AX,DATA_SEG
MOV DS,AX
LEA SI, SRCARY
LEA DI, DESARY
MOV CL,05H
     AGAIN: MOV AL,[SI]
MOV [DI],AL
INC SI
INC DI
DEC CL
JNZ AGAIN
MOV AH,4CH
INT 21H
CODE_SEG ENDS
END START
